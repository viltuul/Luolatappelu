3   import java.util.Random;

5   /**
6    * Luokka on kaikkien olioluokkien yliluokka. Olio sisältää olioiden nimet,
7    * elämät, sijainnin, osumatarkkuuden sekä pelilogiikkaa avittavat attribuutit
8    * elossa, ja Suunnan viimeinenSuunta.
9    */
10  public class Olio {
11  
12      private String nimi;
13      private int elamat;
14      private int x;
15      private int y;
16      private double osumatarkkuus;
17      private boolean elossa;
18      private Suunta viimeinenSuunta;
19  
20      /**
21       * Konstruktori luo olion ja lisää attribuutteihin tarvittavat parametrit.
22       *
23       * @param nimi Oliolle annettava nimi.
24       */
25      public Olio(String nimi) {
26          this.nimi = nimi;
27          this.elamat = 1;
28          this.elossa = true;
29          this.x = 0;
30          this.y = 0;
31          this.osumatarkkuus = 0.5;
32      }
33  
34      public String getNimi() {
35          return nimi;
36      }
37  
38      public void setNimi(String nimi) {
39          this.nimi = nimi;
40      }
41  
42      public int getElamat() {
43          return elamat;
44      }
45  
46      /**
47       * Tarkistaa olion elämät ja palauttaa totuusarvon sen mukaisesti.
48       *
49       * @return true jos olio on elossa
50       */
51      public boolean isElossa() {
52          if (elamat <= 0) {
53              elossa = false;
54          }
55          return elossa;
56      }
57  
58      /**
59       * Metodi vähentää olion elämiä yhdellä.
60       */
61      public void vahennaElamaa() {
62          this.elamat--;
63      }
64  
65      public void setElamat(int elama) {
66          this.elamat = elama;
67      }
68  
69      public int getX() {
70          return x;
71      }
72  
73      public int getY() {
74          return y;
75      }
76  
77      public void setX(int x) {
78          this.x = x;
79      }
80  
81      public void setY(int y) {
82          this.y = y;
83      }
84  
85      /**
86       * Metodi liikuttaa oliota muuttamalla sen x tai y koordinaattia. Metodi
87       * myös tallentaa viimeisen liikkumissuunnan.
88       *
89       * @param suunta Suunta johon olio liikkuu
90       */
91      public void liiku(Suunta suunta) {
92          this.viimeinenSuunta = suunta;
93          if (suunta.equals(suunta.ALAS)) {
94              y++;
95          } else if (suunta.equals(suunta.YLOS)) {
96              y--;
97          } else if (suunta.equals(suunta.VASEN)) {
98              x--;
99          } else if (suunta.equals(suunta.OIKEA)) {
100             x++;
101         }
102     }
103 
104     /**
105      * Jos olio törmääkin toiseen olioon niin liikutetaan Olio takaisin siihen
106      * kohtaan mistä se lähti alunperin.
107      */
108     public void liikuTakaisin() {
109         if (viimeinenSuunta.equals(Suunta.ALAS)) {
110             y--;
111         } else if (viimeinenSuunta.equals(Suunta.YLOS)) {
112             y++;
113         } else if (viimeinenSuunta.equals(Suunta.VASEN)) {
114             x++;
115         } else if (viimeinenSuunta.equals(Suunta.OIKEA)) {
116             x--;
117         }
118     }
119 
120     /**
121      * Metodi käskee oliota lyömään toista oliota jolloin lyötävän elämät
122      * vähenee. Lyönti osuu, jos satunnaismuuttuja on pienempi, kuin olion
123      * osumatarkkuus.
124      *
125      * @param lyotava kertoo Oliolle mitä oliota tulee lyödä.
126      * @return palauttaa true jos olio oosuu lyötävään.
127      */
128     public boolean lyo(Olio lyotava) {
129         if (lyotava != null) {
130             Random osuuko = new Random();
131             if (osuuko.nextDouble() > osumatarkkuus) {
132                 return false;
133             } else {
134                 lyotava.vahennaElamaa();
135                 return true;
136             }
137         }
138         return false;
139     }
140 
141     public double getOsumatarkkuus() {
142         return osumatarkkuus;
143     }
144 
145     public void setOsumatarkkuus(double osumatarkkuus) {
146         this.osumatarkkuus = osumatarkkuus;
147     }
148 
149     /**
150      * Kaikissa olioluokissa on omanlainen liiku metodi jota voidaan kutsua myös
151      * täältä.
152      */
153     public void liiku() {
154     }
