3   import luolatappelu.peli.Peli;

5   /**
6    * Pelaaja on luokan olio aliluokka. Pelaaja luokassa on pelaajan liikkumiseen
7    * vaadittavat metodit.
8    */
9   public class Pelaaja extends Olio {
10  
11      private Peli peli;
12      private int maksimiElamat;
13      private int parannus;
14  
15      /**
16       * Konstruktorissa asetetaan pelaajalle haluttu nimi, ja maksimielämät.
17       *
18       * @param peli peli jossa pelaaja on.
19       */
20      public Pelaaja(Peli peli) {
21          super("Pelaaja");
22          this.peli = peli;
23          this.maksimiElamat = 10;
24          this.parannus = 3;
25          super.setElamat(maksimiElamat);
26      }
27  
28      @Override
29      public void setElamat(int elama) {
30          if (elama > maksimiElamat) {
31              super.setElamat(maksimiElamat);
32          } else {
33              super.setElamat(elama);
34          }
35      }
36  
37      public int getMaksimiElamat() {
38          return maksimiElamat;
39      }
40  
41      public void setMaksimiElamat(int maksimiElamat) {
42          this.maksimiElamat = maksimiElamat;
43      }
44  
45      /**
46       * Metodi kasvattaa maksimielämiä yhdellä.
47       */
48      public void kasvataMaksimia() {
49          maksimiElamat++;
50      }
51  
52      /**
53       * Metodi parantaa pelaajaa parannuksen verran.
54       */
55      public void parannaPelaajaa() {
56          this.setElamat(super.getElamat() + parannus);
57      }
58  
59      /**
60       * Metodi kasvattaa parannusta.
61       */
62      public void kasvataParannusta() {
63          parannus++;
64      }
65  
66      public int getParannus() {
67          return parannus;
68      }
69  
70      public void setParannus(int arvo) {
71          parannus = arvo;
72      }
73  
74      /**
75       * Metodi parantaa pelaajan osumistarkkuutta.
76       */
77      public void kehitaOsumistarkkuutta() {
78          super.setOsumatarkkuus(super.getOsumatarkkuus() + 0.05);
79      }
80  
81      @Override
82      public String toString() {
83          return "@";
84      }
85  
86      /**
87       * Pelaaja toteuttaa metodin lyoTaiLiiku koordinaateilla jotka ovat yhden
88       * askeleen päästä pelaajasta.
89       *
90       * @param suunta Toiminnan suunta pelaajasta katsottuna.
91       */
92      public void toimi(Suunta suunta) {
93          if (suunta.equals(suunta.ALAS)) {
94              lyoTaiLiiku(super.getX(), super.getY() + 1);
95          } else if (suunta.equals(suunta.YLOS)) {
96              lyoTaiLiiku(super.getX(), super.getY() - 1);
97          } else if (suunta.equals(suunta.VASEN)) {
98              lyoTaiLiiku(super.getX() - 1, super.getY());
99          } else if (suunta.equals(suunta.OIKEA)) {
100             lyoTaiLiiku(super.getX() + 1, super.getY());
101         }
102     }
103 
104     /**
105      * Metodi tarkistaa onko annetussa koordinaatissa toista oliota. Jos siinä
106      * on jokin toinen olio, niin pelaaja lyö, jos ei niin pelaaja liikkuu.
107      *
108      * @param x koordinaatti
109      * @param y koordinaatti
110      */
111     public void lyoTaiLiiku(int x, int y) {
112         if (peli.koordinaatinOliot(x, y).isEmpty()) {
113             super.setX(x);
114             super.setY(y);
115         } else {
116             if (super.lyo(peli.koordinaatinOliot(x, y).get(0))) {
117                 peli.getKirjoitin().lyontitapahtuma(this, peli.koordinaatinOliot(x, y).get(0), true);
118             } else {
119                 peli.getKirjoitin().lyontitapahtuma(this, peli.koordinaatinOliot(x, y).get(0), false);
120             }
121         }
122     }
123 
124 }
