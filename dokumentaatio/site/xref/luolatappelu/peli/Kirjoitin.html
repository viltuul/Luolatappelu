1   package luolatappelu.peli;
2   
3   import luolatappelu.objektit.Olio;
4   
5   /**
6    * Kirjoitin on luokka joka kirjoittaa pelin tapahtumat tekstiksi.
7    *
8    * @author ville
9    */
10  public class Kirjoitin {
11  
12      private StringBuilder tapahtuma;
13      private StringBuilder tiedot;
14      private Peli peli;
15      private boolean voimistumisestaIlmoitettu;
16  
17      /**
18       * Konstruktorissa tuodaan vain peliluokka attribuutiksi luokalle.
19       *
20       * @param peli Peli jota pelataan.
21       */
22      public Kirjoitin(Peli peli) {
23          this.peli = peli;
24          this.tapahtuma = new StringBuilder();
25          this.voimistumisestaIlmoitettu = false;
26      }
27  
28      /**
29       * tapahtumienKirjoitin kertoo mitä pelissä kullakin vuorolla on käynyt.
30       *
31       * @return palauttaa tekstin.
32       */
33      public String getTapahtuma() {
34          return tapahtuma.toString();
35      }
36  
37      /**
38       * Kirjoittaa ylös lyöntitapahtuman kutsuttaessa. Jos olio kuolee niin
39       * metodi ilmoittaa myös siitä.
40       *
41       * @param lyoja Olio joka lyö.
42       * @param lyotava Olio jota lyödään.
43       * @param osuiko Totuusarvo joka kertoo osuiko lyönti.
44       */
45      public void lyontitapahtuma(Olio lyoja, Olio lyotava, boolean osuiko) {
46          if (osuiko) {
47              tapahtuma.append(lyoja.getNimi() + " osui olioon " + lyotava.getNimi() + "\n");
48              if (lyotava.getElamat() == 0) {
49                  if (lyotava.toString().equals("P")) {
50                      tapahtuma.append("Olio " + lyotava.getNimi() + " tuhoutui. \n");
51                  } else {
52                      tapahtuma.append("Olio " + lyotava.getNimi() + " kuoli. \n");
53                  }
54              }
55          } else {
56              tapahtuma.append(lyoja.getNimi() + " ei osunut olioon " + lyotava.getNimi() + "\n");
57          }
58      }
59  
60      /**
61       * Metodi ilmoittaa vihollisten voimistumisesta.
62       */
63      public void ilmoitaVihollistenVoimistumisesta() {
64          if (!voimistumisestaIlmoitettu) {
65              tapahtuma.append("\n \n \n \n \n");
66              tapahtuma.append("HUOMIO! \n \n");
67              tapahtuma.append("Viholliset voimistuvat huomattavasti");
68              tapahtuma.append("\n \n \n \n \n");
69              voimistumisestaIlmoitettu = true;
70          }
71      }
72  
73      /**
74       * Tietojen kirjoitin kirjoittaa pelaajan nykyiset elämät, parannuksen ja
75       * osumatarkkuuden.
76       *
77       * @return palauttaa tekstin jossa yllämainitut tiedot.
78       */
79      public String tietojenKirjoitin() {
80          StringBuilder tietojenTulostin = new StringBuilder();
81          tietojenTulostin.append(peli.getPelaaja().getNimi() + "\n");
82          tietojenTulostin.append("Taso: " + peli.getVaikeustaso() + "\n");
83          tietojenTulostin.append("Pelaajan elämät: " + peli.getPelaaja().getElamat() + "/" + peli.getPelaaja().getMaksimiElamat() + "\n");
84          tietojenTulostin.append("Parannus: " + peli.getPelaaja().getParannus() + "\n");
85          tietojenTulostin.append("Osumatarkkuus: " + (int) (peli.getPelaaja().getOsumatarkkuus() * 100) + " % \n");
86          return tietojenTulostin.toString();
87      }
88  
89  }
