48      public void eiMeneReunanYli(Olio olio) {
49          if (olio.isElossa()) {
50              if (olio.getX() > huone.getLeveys() - 1) {
51                  olio.setX(huone.getLeveys() - 2);
52              }
53              if (olio.getY() > huone.getKorkeus() - 1) {
54                  olio.setY(huone.getKorkeus() - 2);
55              }
56              if (olio.getX() < 0) {
57                  olio.setX(0);
58              }
59              if (olio.getY() < 0) {
60                  olio.setY(0);
61              }
62          }
63      }

95      public ArrayList<Olio> getNaapurit(Olio olio) {
96          ArrayList<Olio> lista = new ArrayList();
97          if (koordinaatinOlio(olio.getX() + 1, olio.getY()) != null) {
98              lista.add(koordinaatinOlio(olio.getX() + 1, olio.getY()));
99          }
100         if (koordinaatinOlio(olio.getX() - 1, olio.getY()) != null) {
101             lista.add(koordinaatinOlio(olio.getX() - 1, olio.getY()));
102         }
103         if (koordinaatinOlio(olio.getX(), olio.getY() - 1) != null) {
104             lista.add(koordinaatinOlio(olio.getX(), olio.getY() - 1));
105         }
106         if (koordinaatinOlio(olio.getX(), olio.getY() + 1) != null) {
107             lista.add(koordinaatinOlio(olio.getX(), olio.getY() + 1));
108         }
109         return lista;
