8   import static com.sun.java.accessibility.util.AWTEventMonitor.addActionListener;
9   import java.awt.Container;
10  import java.awt.Dimension;
11  import java.awt.event.ActionEvent;
12  import java.awt.event.ActionListener;
13  import java.util.Scanner;
14  import javax.swing.JFrame;
15  import javax.swing.WindowConstants;
16  import luolatappelu.hahmot.Pelaaja;
17  import luolatappelu.hahmot.Suunta;
18  import luolatappelu.peli.Peli;
19  
20  /**
21   *
22   * @author ville
23   */
24  public class UI {
25  
26      private Scanner lukija;
27      private Peli peli;
28      private Pelaaja pelaaja;
29      private Piirtoalusta alusta;
30      private JFrame frame;
31  
32      public UI(Scanner lukija, Peli peli) {
33          this.lukija = lukija;
34          this.peli = peli;
35          this.pelaaja = peli.getPelaaja();
36          this.alusta = new Piirtoalusta(peli);
37      }
38  
39      public void run() {
40          alku();
41          frame = new JFrame("Luolatappelu");
42          int leveys = (peli.getHuone().getLeveys() * 40 + 30);
43          int korkeus = (peli.getHuone().getKorkeus() * 40 + 30);
44          frame.setPreferredSize(new Dimension(leveys, korkeus));
45          frame.setDefaultCloseOperation(WindowConstants.EXIT_ON_CLOSE);
46          luoKomponentit(frame.getContentPane());
47          frame.pack();
48          frame.setVisible(true);
49      }
50  
51      public void luoKomponentit(Container container) {
52          container.add(alusta);
53          Nappaimistonkuuntelija nk = new Nappaimistonkuuntelija(frame, peli);
54          frame.addKeyListener(nk);
55      }
56  
57      public void alku() {
58          peli.sijoitaPelaaja();
59          peli.sijoitaViholliset();
60      }
61  
62      public JFrame getFrame() {
63          return frame;
64      }
65  
66  
67  }
