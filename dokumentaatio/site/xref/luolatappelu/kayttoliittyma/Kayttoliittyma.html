1   package luolatappelu.kayttoliittyma;
2   
3   import java.awt.Dimension;
4   import java.util.Scanner;
5   import javax.swing.JFrame;
6   import javax.swing.JOptionPane;
7   import javax.swing.WindowConstants;
8   import luolatappelu.peli.Peli;
9   
10  /**
11   * Kayttoliittyma on luokka, joka yhdistää graafisen ilmentymän pelille sekä
12   * pelin kontrollit.
13   */
14  public class Kayttoliittyma implements Runnable {
15  
16      private Scanner lukija;
17      private Peli peli;
18      private Piirtoalusta alusta;
19      private Ruudukko ruudukko;
20      private JFrame peliIkkuna;
21  
22      /**
23       * Konstruktorissa luodaan Peli, Piirtoalusta sekä Ruudukko luokat.
24       *
25       */
26      public Kayttoliittyma() {
27  
28          this.peli = new Peli(this);
29          this.alusta = new Piirtoalusta(peli);
30          this.ruudukko = new Ruudukko();
31      }
32  
33      /**
34       * Metodi luo sekä käynnistää pelin ja kokoaa ikkunan jossa pelaaminen
35       * tapahtuu.
36       */
37      @Override
38      public void run() {
39          peli.uusiTaso();
40          this.peliIkkuna = new JFrame("Luolatappelu");
41          peliIkkuna.setPreferredSize(new Dimension(1300, 870));
42          peliIkkuna.setDefaultCloseOperation(WindowConstants.EXIT_ON_CLOSE);
43          luoKomponentitPeliin();
44          peliIkkuna.pack();
45          peliIkkuna.setVisible(true);
46          kirjoitaNimiIkkuna();
47      }
48  
49      /**
50       * Metodissa luodaan ikkunan komponentit ja kiinnitetään siihen
51       * Nappaimistonkuuntelija.
52       */
53      public void luoKomponentitPeliin() {
54          peliIkkuna.add(ruudukko);
55          ruudukko.setFocusable(false);
56          peliIkkuna.add(alusta);
57          Nappaimistonkuuntelija nk = new Nappaimistonkuuntelija(peliIkkuna, peli);
58          peliIkkuna.addKeyListener(nk);
59      }
60  
61      /**
62       * Metodi kysyy käyttäjältä nimeä, jonka jälkeen asettaa sen pelaajan
63       * nimeksi.
64       */
65      public void kirjoitaNimiIkkuna() {
66          String nimi = JOptionPane.showInputDialog(peliIkkuna, "Kirjoita nimesi");
67          if (nimi != null) {
68              if (!nimi.isEmpty()) {
69                  peli.getPelaaja().setNimi(nimi);
70              }
71          } else {
72              peli.getPelaaja().setNimi("Pelaaja");
73          }
74      }
75  
76      /**
77       * Kun taso päästään läpi niin pelaaja kehittyy ja näytölle ponnahtaa
78       * huomautusikkuna jossa pyydetään pelaajaa valitsemaan haluamansa
79       * kehityskohde.
80       */
81      public void kehitysIkkuna() {
82          Object[] vaihtoehdot = {"Elämiä", "Parantuvuutta", "Osumistarkkuutta"};
83          int vaihtoehtoNro = JOptionPane.showOptionDialog(peliIkkuna,
84                  "Valitse yksi",
85                  "Kehityit",
86                  JOptionPane.YES_NO_CANCEL_OPTION,
87                  JOptionPane.QUESTION_MESSAGE,
88                  null,
89                  vaihtoehdot,
90                  vaihtoehdot[0]);
91  
92          if (vaihtoehtoNro == 2) {
93              peli.getPelaaja().kehitaOsumistarkkuutta();
94          } else if (vaihtoehtoNro == 1) {
95              peli.getPelaaja().kasvataParannusta();
96          } else {
97              peli.getPelaaja().kasvataMaksimia();
98          }
99      }
100 
101     /**
102      * peliOhi metodi lopettaa pelin ja aloittaa sen alusta.
103      */
104     public void peliOhi() {
105         JOptionPane.showMessageDialog(null, "Kuolit, peli alkaa alusta uudestaan");
106         peliIkkuna.dispose();
107         peli.nollaaKaikki();
108         run();
109     }
110 
111     /**
112      * Metodi ilmoittaa pelin olevan läpipelattu ja sammuttaa pelin.
113      */
114     public void peliLapi() {
115         JOptionPane.showMessageDialog(null, "Pääsit pelin läpi, mene nyt ulos leikkimään!");
116         peliIkkuna.dispose();
117     }
118 
119     public Ruudukko getRuudukko() {
120         return ruudukko;
121     }
122 
123 }
