3   import java.awt.Color;
4   import java.awt.Graphics;
5   import javax.swing.JPanel;
6   import luolatappelu.objektit.Olio;
7   import luolatappelu.objektit.Oviaukko;
8   import luolatappelu.objektit.Pelaaja;
9   import luolatappelu.peli.Peli;

11  /**
12   * Piirtoalustassa luodaan kaikki pelin graafiset ominaisuudet.
13   */
14  public class Piirtoalusta extends JPanel {
15  
16      private Peli peli;
17      private Graphics grafiikka;
18  
19      /**
20       * Asetetaan piirtoalustalle harmaa taustav채ri sek채 asetetaan rajat.
21       *
22       * @param peli Peli kiinnitet채채n attribuuttiin.
23       */
24      public Piirtoalusta(Peli peli) {
25          super.setBackground(Color.GRAY);
26          this.peli = peli;
27          super.setBounds(0, 0, 840, 870);
28      }
29  
30      @Override
31      public void paintComponent(Graphics grafiikka) {
32          this.grafiikka = grafiikka;
33          super.paintComponent(grafiikka);
34          piirraPelaaja();
35          piirraSeuraajat();
36          piirraOrkit();
37          piirraTankit();
38          piirraSeinat();
39          piirraOviaukko();
40          piirraPuuseinat();
41      }
42  
43      private void piirraPelaaja() {
44          Pelaaja pelaaja = peli.getPelaaja();
45          piirraVartalo(pelaaja, Color.BLUE);
46          this.piirraNaama(pelaaja.getX(), pelaaja.getY(), false);
47          piirraMiekka(pelaaja.getX() * 40, pelaaja.getY() * 40);
48      }
49  
50      private void piirraSeuraajat() {
51          for (Olio olio : peli.getOliokanta().getSeuraajat()) {
52              piirraVartalo(olio, Color.RED);
53              this.piirraNaama(olio.getX(), olio.getY(), true);
54              piirraKeihas(olio.getX() * 40, olio.getY() * 40);
55          }
56      }
57  
58      private void piirraOrkit() {
59          for (Olio olio : peli.getOliokanta().getOrkit()) {
60              piirraVartalo(olio, Color.GREEN);
61              this.piirraNaama(olio.getX(), olio.getY(), false);
62              piirraMiekka(olio.getX() * 40, olio.getY() * 40);
63          }
64      }
65  
66      private void piirraTankit() {
67          for (Olio olio : peli.getOliokanta().getTankit()) {
68              piirraVartalo(olio, Color.YELLOW);
69              this.piirraNaama(olio.getX(), olio.getY(), true);
70              this.piirraKilpi(olio.getX() * 40, olio.getY() * 40);
71          }
72      }
73  
74      private void piirraVartalo(Olio olio, Color vari) {
75          grafiikka.setColor(new Color(139, 69, 19));
76          grafiikka.fillOval(olio.getX() * 40, olio.getY() * 40 + 27, 12, 8);
77          grafiikka.fillOval(olio.getX() * 40 + 20, olio.getY() * 40 + 27, 12, 8);
78          grafiikka.setColor(vari);
79          grafiikka.fillOval(olio.getX() * 40, olio.getY() * 40, 30, 30);
80          grafiikka.fillOval(olio.getX() * 40 - 10, olio.getY() * 40 + 5, 15, 15);
81          grafiikka.fillOval(olio.getX() * 40 + 20, olio.getY() * 40 + 5, 15, 15);
82          grafiikka.setColor(Color.BLACK);
83          grafiikka.drawOval(olio.getX() * 40, olio.getY() * 40, 30, 30);
84  
85      }
86  
87      private void piirraNaama(int x, int y, Boolean vihainen) {
88          grafiikka.setColor(Color.BLACK);
89          grafiikka.fillRect(x * 40 + 7, y * 40 + 7, 4, 12);
90          grafiikka.fillRect(x * 40 + 18, y * 40 + 7, 4, 12);
91          grafiikka.fillRect(x * 40 + 10, y * 40 + 23, 12, 4);
92          grafiikka.fillRect(x * 40 + 8, y * 40 + 21, 4, 4);
93          grafiikka.fillRect(x * 40 + 20, y * 40 + 21, 4, 4);
94          if (vihainen) {
95              grafiikka.fillRect(x * 40 + 5, y * 40 + 5, 4, 4);
96              grafiikka.fillRect(x * 40 + 20, y * 40 + 5, 4, 4);
97          }
98      }
99  
100     private void piirraKeihas(int x, int y) {
101         grafiikka.setColor(new Color(139, 69, 0));
102         grafiikka.fillRect(x - 10, y + 5, 3, 30);
103         grafiikka.setColor(Color.darkGray);
104         grafiikka.fillOval(x - 11, y - 2, 5, 10);
105     }
106 
107     private void piirraMiekka(int x, int y) {
108         grafiikka.setColor(new Color(139, 69, 0));
109         grafiikka.fill3DRect(x - 10, y + 8, 10, 6, true);
110         grafiikka.setColor(Color.darkGray);
111         grafiikka.fillOval(x - 8, y - 5, 5, 15);
112     }
113 
114     private void piirraKilpi(int x, int y) {
115         grafiikka.setColor(Color.darkGray);
116         grafiikka.fillRect(x + 6, y + 13, 30, 15);
117         grafiikka.setColor(Color.black);
118         grafiikka.drawRect(x + 6, y + 13, 30, 15);
119         grafiikka.setColor(Color.gray);
120         grafiikka.fillOval(x + 6, y + 15, 30, 20);
121         grafiikka.setColor(Color.BLACK);
122         grafiikka.drawOval(x + 6, y + 15, 30, 20);
123         grafiikka.setColor(Color.yellow);
124         grafiikka.fillOval(x + 18, y + 18, 4, 4);
125         grafiikka.fillOval(x + 24, y + 18, 4, 4);
126         grafiikka.fillOval(x + 21, y + 24, 4, 4);
127 
128     }
129 
130     private void piirraSeinat() {
131         for (Olio seina : peli.getTaso().getLuolasto().getSeinat()) {
132             grafiikka.setColor(Color.black);
133             grafiikka.fillRect(seina.getX() * 40, seina.getY() * 40, 40, 40);
134             for (int i = 0; i < 4; i++) {
135                 grafiikka.setColor(Color.darkGray);
136                 grafiikka.fillOval(seina.getX() * 40, seina.getY() * 40 + (i * 10), 10, 5);
137                 grafiikka.fillOval(seina.getX() * 40 + 20, seina.getY() * 40 + (i * 10), 10, 5);
138                 grafiikka.fillOval(seina.getX() * 40 + 30, seina.getY() * 40 + (i * 10 + 5), 10, 5);
139             }
140         }
141     }
142 
143     private void piirraOviaukko() {
144         Oviaukko ovi = peli.getTaso().getOvi();
145         if (peli.getTaso().isTasoLapi()) {
146             grafiikka.setColor(Color.black);
147         } else {
148             grafiikka.setColor(Color.gray);
149         }
150         grafiikka.fillOval(ovi.getX() * 40, ovi.getY() * 40, 40, 40);
151 
152     }
153 
154     private void piirraPuuseinat() {
155         for (Olio olio : peli.getOliokanta().getPuuseinat()) {
156             grafiikka.setColor(new Color(210, 105, 30));
157             grafiikka.fillRect(olio.getX() * 40, olio.getY() * 40, 40, 40);
158         }
159     }
