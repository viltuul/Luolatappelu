1   package luolatappelu.hahmot;
2   
3   import luolatappelu.Suunta;
4   import luolatappelu.peli.Peli;
5   
6   /**
7    * Pelaaja on luokan olio aliluokka. Pelaaja luokassa on pelaajan liikkumiseen
8    * vaadittavat metodit.
9    */
10  public class Pelaaja extends Olio {
11  
12      private Peli peli;
13  
14      public Pelaaja(String nimi, Peli peli) {
15          super(nimi);
16          super.setElamat(10);
17          this.peli = peli;
18      }
19  
20      @Override
21      public String toString() {
22          return "@";
23      }
24  
25      public void toimi(Suunta suunta) {
26          if (suunta.equals(suunta.ALAS)) {
27              lyoTaiLiiku(super.getX(), super.getY() + 1);
28          } else if (suunta.equals(suunta.YLOS)) {
29              lyoTaiLiiku(super.getX(), super.getY() - 1);
30          } else if (suunta.equals(suunta.VASEN)) {
31              lyoTaiLiiku(super.getX() - 1, super.getY());
32          } else if (suunta.equals(suunta.OIKEA)) {
33              lyoTaiLiiku(super.getX() + 1, super.getY());
34          }
35      }
36  
37      public void lyoTaiLiiku(int x, int y) {
38          if (peli.koordinaatinOliot(x, y).size() == 0) {
39              super.setX(x);
40              super.setY(y);
41          } else {
42              super.lyo(peli.koordinaatinOliot(x, y).get(0));
43          }
44      }
45  
46  }
